229 template <typename T>
230 void emplace(T&& value) {
231 using U = std::decay_t<T>;
232 const VTable* vt = makeVTable<U>();
233 reset();
234
235 type = &typeid(U);
236 vtable = vt;
237
238 if (sizeof(U) <= SBO_SIZE && alignof(U) <= alignof(AlignedBuffer)) {
239 onHeap = false;
240 new (&sboBuf) U(std::forward<T>(value));
241 } else {
242 onHeap = true;
243 heapPtr = operator new(sizeof(U));
244 new (heapPtr) U(std::forward<T>(value));
245 }
246 }

254 void reset() noexcept {
255 if (vtable) {
256 vtable->destroy(dataPtr());
257 if (onHeap) {
258 operator delete(heapPtr);
259 }
260 vtable = nullptr;
261 type = nullptr;
262 }
