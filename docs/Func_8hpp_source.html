18 protected:
19 static constexpr Size bufSize = sizeof(void*) * 8;
20 using Buffer = std::aligned_storage_t<bufSize, alignof(std::max_align_t)>;
21
22 struct CallableBase {
23 virtual R invoke(Args&&...) = 0;
24 virtual CallableBase* clone(Buffer& buffer) const = 0;
25 virtual CallableBase* move(Buffer& buffer) = 0;
26 virtual void destroy(bool inBuffer) noexcept = 0;
27 virtual ~CallableBase() = default;
28 };

30 template <typename F>
31 struct CallableImpl: CallableBase {
32 F VaFunc;
33
34 template <typename U>
35 CallableImpl(U&& f) : VaFunc(std::forward<U>(f)) {}
36
37 R invoke(Args&&... args) override { return VaFunc(std::forward<Args>(args)...); }
38
39 CallableBase* clone(Buffer& buffer) const override {
40 if (sizeof(CallableImpl) <= bufSize) {
41 return new (&buffer) CallableImpl(VaFunc);
42 } else {
43 return new CallableImpl(VaFunc);
44 }
45 }
46
47 CallableBase* move(Buffer& buffer) override {
48 if (sizeof(CallableImpl) <= bufSize) {
49 return new (&buffer) CallableImpl(std::move(VaFunc));
50 } else {
51 return new CallableImpl(std::move(VaFunc));
52 }
53 }
54
55 void destroy(bool inBuffer) noexcept override {
56 if (inBuffer) {
57 this->~CallableImpl();
58 } else {
59 delete this;
60 }
61 }
62 };
