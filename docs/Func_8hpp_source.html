20 struct alignas(MaxAlignType) Buffer {

24 struct CallableBase {
25 virtual R invoke(Args&&...) = 0;
26 virtual CallableBase* clone(Buffer& buffer) const = 0;
27 virtual CallableBase* move(Buffer& buffer) = 0;
28 virtual void destroy(bool inBuffer) = 0;
29 virtual ~CallableBase() = default;
30 };

32 template <typename F>
33 struct CallableImpl : CallableBase {
34 F func;
35
36 template <typename U>
37 CallableImpl(U&& f) : func(std::forward<U>(f)) {}
38
39 R invoke(Args&&... args) override {
40 return func(std::forward<Args>(args)...);
41 }
42
43 CallableBase* clone(Buffer& buffer) const override {
44 if (sizeof(CallableImpl) <= bufSize) {
45 return new (&buffer) CallableImpl(func);
46 } else {
47 return new CallableImpl(func);
48 }
49 }
50
51 CallableBase* move(Buffer& buffer) override {
52 if (sizeof(CallableImpl) <= bufSize) {
53 return new (&buffer) CallableImpl(std::move(func));
54 } else {
55 return new CallableImpl(std::move(func));
56 }
57 }
58
59 void destroy(bool inBuffer) noexcept(tt::IsNoexceptDestructible<F>) override {
60 if (inBuffer) {
61 this->~CallableImpl();
62 } else {
63 delete this;
64 }
65 }
66 };
