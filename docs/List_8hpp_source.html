36 void resize(Size newCap) {
37 T* newData = static_cast<T*>(std::malloc(newCap * sizeof(T)));

40 if constexpr (std::is_trivially_copyable_v<T>) {

275 void insert(Size index, T value) {
276 if (index > len) throw IndexOutOfRangeError(len, index);
277 update();
278 for (Size i = len; i > index; i--) {
279 new (&data[i]) T(std::move(data[i - 1]));
280 data[i - 1].~T();
281 }
282 new (&data[index]) T(std::move(value));
283 len++;
284 }

293 void insertList(Size index, const VaList& other) {
294 if (index > len) throw IndexOutOfRangeError(len, index);
295 if (other.len == 0) return;
296
297 Size newLen = len + other.len;
298 if (newLen > cap) resize(newLen);
299
300 for (Size i = len - 1; i >= index && i != (Size)-1; i--) {
301 new (&data[i + other.len]) T(std::move(data[i]));
302 data[i].~T();
303 }
304
305 for (Size i = 0; i < other.len; i++) {
306 new (&data[index + i]) T(other.data[i]);
307 }
308
309 len = newLen;
310 }
