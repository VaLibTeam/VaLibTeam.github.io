70 void resize(Size newCap) {
71 Entry** newBuckets = new Entry*[newCap]();
72 for (Size i = 0; i < newCap; ++i) {
73 newBuckets[i] = nullptr;
74 }
75
76 Entry* current = head;
77 while (current != nullptr) {
78 Size index = hashFunc(current->key) % newCap;
79 Entry* entry = current;
80 entry->next = newBuckets[index];
81 newBuckets[index] = entry;
82 current = current->nextOrder;
83 }
84
85 delete[] buckets;
86 buckets = newBuckets;
87 cap = newCap;
88 }

95 void insertBefore(Entry* next, Entry* newEntry) {
96 Entry* prev = next ? next->prevOrder : tail;
97
98 newEntry->nextOrder = next;
99 newEntry->prevOrder = prev;
100
101 if (prev) {
102 prev->nextOrder = newEntry;
103 } else {
104 head = newEntry;
105 }
106
107 if (next) {
108 next->prevOrder = newEntry;
109 } else {
110 tail = newEntry;
111 }

118 void appendEntry(Entry* newEntry) {
119 newEntry->prevOrder = tail;

136 Entry* removeFromBucket(Size index, const K& key) {
137 Entry* entry = buckets[index];
138 Entry* prev = nullptr;
139
140 while (entry) {
141 if (entry->key == key) {
142 if (prev) {
143 prev->next = entry->next;
144 } else {
145 buckets[index] = entry->next;
146 }
147 return entry;
148 }
149
150 prev = entry;
151 entry = entry->next;
152 }
153
154 return nullptr;
155 }

161 void unlinkEntry(Entry* entry) {
162 if (entry->prevOrder) {
163 entry->prevOrder->nextOrder = entry->nextOrder;
164 } else {
165 head = entry->nextOrder;
166 }
167
168 if (entry->nextOrder) {
169 entry->nextOrder->prevOrder = entry->prevOrder;
170 } else {
171 tail = entry->prevOrder;
172 }
173
174 entry->nextOrder = entry->prevOrder = nullptr;
175 }

475 Size hash = 0;

477 while (current != nullptr) {

479 current = current->nextOrder;

505 void insert(Size index, const K& key, const V& value) {
506 if (index > size) throw IndexOutOfRangeError(size, index);
507
508 if (contains(key)) {
509 del(key);
510 if (index > size) index = size;
511 }
512
513 if (size >= cap * 0.75) resize(cap * 2);
514
515 Size bucketIndex = computeIndex(key);
516 Entry* newEntry = new Entry(key, value);
517
518 newEntry->next = buckets[bucketIndex];
519 buckets[bucketIndex] = newEntry;
520
521 if (index == size) {
522 appendEntry(newEntry);
523 } else {
524 Entry* target = head;
525 for (Size i = 0; i < index; ++i) {
526 target = target->nextOrder;
527 }
528 insertBefore(target, newEntry);
529 }
530
531 size++;
532 }
