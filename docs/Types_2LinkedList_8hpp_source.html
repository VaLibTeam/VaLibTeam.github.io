42 void unlinkFromOrder(Node* node) noexcept {
43 if (node->prev) {
44 node->prev->next = node->next;
45 } else {
46 head = node->next;
47 }
48
49 if (node->next) {
50 node->next->prev = node->prev;
51 } else {
52 tail = node->prev;
53 }
54
55 node->next = nullptr;
56 node->prev = nullptr;
57 }

63 void linkToEnd(Node* node) noexcept {
64 node->prev = tail;
65 node->next = nullptr;
66
67 if (tail) {
68 tail->next = node;
69 } else {
70 head = node;
71 }
72
73 tail = node;
74 }

185 Node* nodeAt(Size index) const {
186 if (index < len / 2) {
187 Node* current = head;
188 for (Size i = 0; i < index; ++i) current = current->next;
189 return current;
190 } else {
191 Node* current = tail;
192 for (Size i = len - 1; i > index; --i) current = current->prev;
193 return current;
194 }

700 void insertEach(Size pos, VaLinkedList&& other) {
701 if (other.len == 0 || this == &other) return;
702 if (pos > len) throw IndexOutOfRangeError(len, pos);
703 if (pos == len) {
704 appendEach(std::move(other));
705 return;

708 prependEach(std::move(other));

715 other.head->prev = before;
716 other.tail->next = atNode;
717 atNode->prev = other.tail;
718

720 other.head = other.tail = nullptr;
721 other.len = 0;

1042 friend bool operator==(const VaLinkedList& lhs, const VaLinkedList& rhs) {
1043 if (&lhs == &rhs) return true;
1044 if (lhs.len != rhs.len) return false;
1045
1046 Node* a = lhs.head;
1047 Node* b = rhs.head;
1048
1049 while (a && b) {
1050 #if __cplusplus >= CPP17
1051 if constexpr (tt::HasEqualityOperator_v<T>) {
1052 if (!(a->value == b->value)) return false;
1053 } else if constexpr (tt::IsTriviallyCopyable<T>) {
1054 if (std::memcmp(&a->value, &b->value, sizeof(T)) != 0) return false;
1055 } else {
1056 static_assert(sizeof(T) == -1, "Cannon compare list with non-comparable elements");
1057 }
1058 #else
1059 if (!(a->value == b->value)) return false;
1060 #endif
1061
1062 a = a->next;
1063 b = b->next;
1064 }
1065
1066 return a == nullptr && b == nullptr;
1067 }
